var jsonData = pm.response.json()
const role = jsonData.data.id
pm.environment.set("roleId", role)

pm.test("Status code is 201", function(){
    pm.response.to.have.status(201);
})
pm.test("Status code is OK", function () {
    pm.response.to.have.status("Created");
});

pm.test("Statu is success", function(){
    pm.expect(jsonData.status).to.eql("success")
})
pm.test("Error message should be 'Unauthorized'", function(){
    pm.expect(jsonData.message).to.eql("Role created successfully")
})
